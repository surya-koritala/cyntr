> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 7 features to Cyntr: webhook integrations (PagerDuty/Datadog), SLA monitoring, scheduled reports, real token streaming, agent templates marketplace, custom tool builder UI, and observability dashboard.

**Architecture:** Each feature is a standalone module or extension. Webhook integrations extend `modules/notify/`. SLA monitoring is a new kernel module subscribing to `agent.activity` events. Scheduled reports extend the existing scheduler. Token streaming uses the existing `StreamingProvider` interface. Agent templates, tool builder, and observability are API + dashboard additions.

**Tech Stack:** Go 1.26, SQLite, SSE, HTTP webhooks, existing kernel/IPC patterns.

```go
// modules/notify/pagerduty.go
package notify

import (
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

// PagerDutyChannel sends notifications to PagerDuty Events API v2.
type PagerDutyChannel struct {
    endpoint   string
    routingKey string
    client     *http.Client
}

func NewPagerDutyChannel(endpoint, routingKey string) *PagerDutyChannel {
    if endpoint == "" {
        endpoint = "https://events.pagerduty.com/v2/enqueue"
    }
    return &PagerDutyChannel{
        endpoint:   endpoint,
        routingKey: routingKey,
        client:     &http.Client{Timeout: 15 * time.Second},
    }
}

func (c *PagerDutyChannel) Name() string { return "pagerduty" }

func (c *PagerDutyChannel) Send(ctx context.Context, n Notification) error {
    action := "trigger"
    if n.Severity == "info" || n.Severity == "resolved" {
        action = "resolve"
    }

    severity := n.Severity
    switch severity {
    case "critical", "error":
        severity = "critical"
    case "warning":
        severity = "warning"
    default:
        severity = "info"
    }

    payload := map[string]any{
        "routing_key":  c.routingKey,
        "event_action": action,
        "dedup_key":    fmt.Sprintf("cyntr-%s-%s-%s", n.Tenant, n.Agent, n.Source),
        "payload": map[string]any{
            "summary":   n.Title + ": " + n.Message,
            "severity":  severity,
            "source":    "cyntr/" + n.Tenant,
            "component": n.Agent,
            "group":     n.Tenant,
            "custom_details": map[string]string{
                "agent":  n.Agent,
                "tenant": n.Tenant,
                "source": n.Source,
            },
        },
    }

    body, _ := json.Marshal(payload)
    req, err := http.NewRequestWithContext(ctx, "POST", c.endpoint, bytes.NewReader(body))
    if err != nil {
        return fmt.Errorf("pagerduty: create request: %w", err)
    }
    req.Header.Set("Content-Type", "application/json")

    resp, err := c.client.Do(req)
    if err != nil {
        return fmt.Errorf("pagerduty: send: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode >= 400 {
        return fmt.Errorf("pagerduty: HTTP %d", resp.StatusCode)
    }
    return nil
}
```

```go
// modules/notify/datadog.go
package notify

import (
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

// DatadogChannel sends events and metrics to Datadog API.
type DatadogChannel struct {
    endpoint string
    apiKey   string
    client   *http.Client
}

func NewDatadogChannel(endpoint, apiKey string) *DatadogChannel {
    if endpoint == "" {
        endpoint = "https://api.datadoghq.com"
    }
    return &DatadogChannel{
        endpoint: endpoint,
        apiKey:   apiKey,
        client:   &http.Client{Timeout: 15 * time.Second},
    }
}

func (c *DatadogChannel) Name() string { return "datadog" }

func (c *DatadogChannel) Send(ctx context.Context, n Notification) error {
    alertType := "info"
    switch n.Severity {
    case "critical":
        alertType = "error"
    case "warning":
        alertType = "warning"
    case "error":
        alertType = "error"
    }

    event := map[string]any{
        "title":      n.Title,
        "text":       n.Message,
        "alert_type": alertType,
        "source_type_name": "cyntr",
        "tags": []string{
            "agent:" + n.Agent,
            "tenant:" + n.Tenant,
            "source:" + n.Source,
        },
    }

    return c.post(ctx, "/api/v1/events", event)
}

func (c *DatadogChannel) SendMetric(ctx context.Context, metric string, value float64, tags map[string]string) error {
    var tagList []string
    for k, v := range tags {
        tagList = append(tagList, k+":"+v)
    }

    payload := map[string]any{
        "series": []map[string]any{{
            "metric": metric,
            "points": [][]any{{float64(time.Now().Unix()), value}},
            "type":   "gauge",
            "tags":   tagList,
        }},
    }

    return c.post(ctx, "/api/v1/series", payload)
}

func (c *DatadogChannel) post(ctx context.Context, path string, payload any) error {
    body, _ := json.Marshal(payload)
    req, err := http.NewRequestWithContext(ctx, "POST", c.endpoint+path, bytes.NewReader(body))
    if err != nil {
        return fmt.Errorf("datadog: create request: %w", err)
    }
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("DD-API-KEY", c.apiKey)

    resp, err := c.client.Do(req)
    if err != nil {
        return fmt.Errorf("datadog: send: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode >= 400 {
        return fmt.Errorf("datadog: HTTP %d", resp.StatusCode)
    }
    return nil
}
```

```go
// modules/notify/generic_webhook.go
package notify

import (
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "time"
)

// GenericWebhookChannel sends JSON notifications to any HTTP endpoint.
type GenericWebhookChannel struct {
    name         string
    url          string
    headers      map[string]string
    client       *http.Client
    MaxRetries   int
    RetryDelayMs int
}

func NewGenericWebhookChannel(name, url string, headers map[string]string) *GenericWebhookChannel {
    return &GenericWebhookChannel{
        name:         name,
        url:          url,
        headers:      headers,
        client:       &http.Client{Timeout: 15 * time.Second},
        MaxRetries:   1,
        RetryDelayMs: 1000,
    }
}

func (c *GenericWebhookChannel) Name() string { return c.name }

func (c *GenericWebhookChannel) Send(ctx context.Context, n Notification) error {
    payload := map[string]any{
        "title":     n.Title,
        "message":   n.Message,
        "severity":  n.Severity,
        "agent":     n.Agent,
        "tenant":    n.Tenant,
        "source":    n.Source,
        "timestamp": time.Now().UTC().Format(time.RFC3339),
    }

    body, _ := json.Marshal(payload)

    var lastErr error
    for attempt := 0; attempt < c.MaxRetries; attempt++ {
        if attempt > 0 {
            time.Sleep(time.Duration(c.RetryDelayMs) * time.Millisecond)
        }

        req, err := http.NewRequestWithContext(ctx, "POST", c.url, bytes.NewReader(body))
        if err != nil {
            return fmt.Errorf("webhook %s: create request: %w", c.name, err)
        }
        req.Header.Set("Content-Type", "application/json")
        for k, v := range c.headers {
            req.Header.Set(k, v)
        }

        resp, err := c.client.Do(req)
        if err != nil {
            lastErr = fmt.Errorf("webhook %s: send: %w", c.name, err)
            continue
        }
        resp.Body.Close()

        if resp.StatusCode < 400 {
            return nil
        }
        lastErr = fmt.Errorf("webhook %s: HTTP %d", c.name, resp.StatusCode)
    }

    return lastErr
}
```

```go
// modules/sla/types.go
package sla

import "time"

// Rule defines an SLA threshold for an agent.
type Rule struct {
    ID              string        `json:"id"`
    Agent           string        `json:"agent"`            // agent name or "*" for all
    Tenant          string        `json:"tenant"`           // tenant or "*"
    MaxResponseMs   int64         `json:"max_response_ms"`  // max allowed response time
    MaxErrorRate    float64       `json:"max_error_rate"`   // max error percentage (0-100)
    WindowMinutes   int           `json:"window_minutes"`   // evaluation window (default 5)
    NotifyChannels  []string      `json:"notify_channels"`  // channels to alert on breach
    Enabled         bool          `json:"enabled"`
}

// Violation records when an SLA was breached.
type Violation struct {
    ID         string    `json:"id"`
    RuleID     string    `json:"rule_id"`
    Agent      string    `json:"agent"`
    Tenant     string    `json:"tenant"`
    Type       string    `json:"type"`     // "latency" or "error_rate"
    Value      float64   `json:"value"`    // actual value
    Threshold  float64   `json:"threshold"` // SLA threshold
    Timestamp  time.Time `json:"timestamp"`
    Resolved   bool      `json:"resolved"`
    ResolvedAt time.Time `json:"resolved_at,omitempty"`
}
```

```go
import (
    "context"
    "fmt"
    "sync"
    "time"

    "github.com/cyntr-dev/cyntr/kernel"
    "github.com/cyntr-dev/cyntr/kernel/ipc"

    "github.com/cyntr-dev/cyntr/modules/notify"
)

var logger = log.Default().WithModule("sla")

type latencyRecord struct {
    DurationMs int64
    Timestamp  time.Time
}

type Monitor struct {
    mu          sync.RWMutex
    bus         *ipc.Bus
    notifier    *notify.Notifier
    rules       []Rule
    violations  []Violation
    latencies   map[string][]latencyRecord // "tenant/agent" -> records
    errors      map[string][]time.Time     // "tenant/agent" -> error timestamps
    counter     int64
}

func New() *Monitor {
    return &Monitor{
        latencies: make(map[string][]latencyRecord),
        errors:    make(map[string][]time.Time),
    }
}

func (m *Monitor) Name() string           { return "sla" }
func (m *Monitor) Dependencies() []string { return []string{"agent_runtime"} }

func (m *Monitor) SetNotifier(n *notify.Notifier) { m.notifier = n }

func (m *Monitor) Init(ctx context.Context, svc *kernel.Services) error {
    m.bus = svc.Bus
    return nil
}

func (m *Monitor) Start(ctx context.Context) error {
    m.bus.Handle("sla", "sla.add_rule", m.handleAddRule)
    m.bus.Handle("sla", "sla.list_rules", m.handleListRules)
    m.bus.Handle("sla", "sla.remove_rule", m.handleRemoveRule)
    m.bus.Handle("sla", "sla.violations", m.handleListViolations)

    // Subscribe to agent activity events
    m.bus.Subscribe("sla", "agent.activity", func(msg ipc.Message) (ipc.Message, error) {
        if evt, ok := msg.Payload.(agent.ActivityEvent); ok {
            m.processEvent(evt)
        }
        return ipc.Message{}, nil
    })

    // Start background SLA checker
    go m.runChecker(ctx)

    return nil
}

func (m *Monitor) Stop(ctx context.Context) error { return nil }

func (m *Monitor) Health(ctx context.Context) kernel.HealthStatus {
    m.mu.RLock()
    defer m.mu.RUnlock()
    active := 0
    for _, r := range m.rules {
        if r.Enabled { active++ }
    }
    return kernel.HealthStatus{
        Healthy: true,
        Message: fmt.Sprintf("%d rules, %d violations", active, len(m.violations)),
    }
}

func (m *Monitor) AddRule(rule Rule) {
    m.mu.Lock()
    defer m.mu.Unlock()
    if rule.WindowMinutes == 0 {
        rule.WindowMinutes = 5
    }
    m.rules = append(m.rules, rule)
}

func (m *Monitor) RecordResponse(agentName, tenant string, durationMs int64) {
    m.mu.Lock()
    defer m.mu.Unlock()
    key := tenant + "/" + agentName
    m.latencies[key] = append(m.latencies[key], latencyRecord{
        DurationMs: durationMs, Timestamp: time.Now(),
    })
}

func (m *Monitor) RecordError(agentName, tenant string) {
    m.mu.Lock()
    defer m.mu.Unlock()
    key := tenant + "/" + agentName
    m.errors[key] = append(m.errors[key], time.Now())
}

func (m *Monitor) Check() []Violation {
    m.mu.Lock()
    defer m.mu.Unlock()

    now := time.Now()
    var newViolations []Violation

    for _, rule := range m.rules {
        if !rule.Enabled {
            continue
        }
        window := time.Duration(rule.WindowMinutes) * time.Minute
        key := rule.Tenant + "/" + rule.Agent

        // Check latency
        if rule.MaxResponseMs > 0 {
            records := m.latencies[key]
            var recent []latencyRecord
            var totalMs int64
            for _, r := range records {
                if now.Sub(r.Timestamp) <= window {
                    recent = append(recent, r)
                    totalMs += r.DurationMs
                }
            }
            m.latencies[key] = recent // prune old

            if len(recent) > 0 {
                avgMs := totalMs / int64(len(recent))
                if avgMs > rule.MaxResponseMs {
                    m.counter++
                    newViolations = append(newViolations, Violation{
                        ID: fmt.Sprintf("vio_%d", m.counter),
                        RuleID: rule.ID, Agent: rule.Agent, Tenant: rule.Tenant,
                        Type: "latency", Value: float64(avgMs),
                        Threshold: float64(rule.MaxResponseMs),
                        Timestamp: now,
                    })
                }
            }
        }
    }

    m.violations = append(m.violations, newViolations...)
    return newViolations
}

func (m *Monitor) runChecker(ctx context.Context) {
    ticker := time.NewTicker(30 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            return
        case <-ticker.C:
            violations := m.Check()
            for _, v := range violations {
                logger.Warn("SLA violation", map[string]any{
                    "rule": v.RuleID, "agent": v.Agent, "type": v.Type,
                    "value": v.Value, "threshold": v.Threshold,
                })
                if m.notifier != nil {
                    m.notifier.Send(context.Background(), notify.Notification{
                        Title:    fmt.Sprintf("SLA Breach: %s/%s", v.Tenant, v.Agent),
                        Message:  fmt.Sprintf("%s: %.0f (threshold: %.0f)", v.Type, v.Value, v.Threshold),
                        Severity: "warning",
                        Agent:    v.Agent,
                        Tenant:   v.Tenant,
                        Source:   "sla-monitor",
                    })
                }
            }
        }
    }
}

func (m *Monitor) handleAddRule(msg ipc.Message) (ipc.Message, error) {
    rule, ok := msg.Payload.(Rule)
    if !ok {
        return ipc.Message{}, fmt.Errorf("expected Rule, got %T", msg.Payload)
    }
    m.AddRule(rule)
    return ipc.Message{Type: ipc.MessageTypeResponse, Payload: "ok"}, nil
}

func (m *Monitor) handleListRules(msg ipc.Message) (ipc.Message, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    rules := make([]Rule, len(m.rules))
    copy(rules, m.rules)
    return ipc.Message{Type: ipc.MessageTypeResponse, Payload: rules}, nil
}

func (m *Monitor) handleRemoveRule(msg ipc.Message) (ipc.Message, error) {
    ruleID, _ := msg.Payload.(string)
    m.mu.Lock()
    defer m.mu.Unlock()
    for i, r := range m.rules {
        if r.ID == ruleID {
            m.rules = append(m.rules[:i], m.rules[i+1:]...)
            return ipc.Message{Type: ipc.MessageTypeResponse, Payload: "removed"}, nil
        }
    }
    return ipc.Message{}, fmt.Errorf("rule %q not found", ruleID)
}

func (m *Monitor) handleListViolations(msg ipc.Message) (ipc.Message, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    v := make([]Violation, len(m.violations))
    copy(v, m.violations)
    return ipc.Message{Type: ipc.MessageTypeResponse, Payload: v}, nil
}
```

```go
// web/api/sla.go
package api

import (
    "context"
    "encoding/json"
    "net/http"
    "time"

    "github.com/cyntr-dev/cyntr/kernel/ipc"
    "github.com/cyntr-dev/cyntr/modules/sla"
)

func (s *Server) handleSLAAddRule(w http.ResponseWriter, r *http.Request) {
    var rule sla.Rule
    if err := json.NewDecoder(r.Body).Decode(&rule); err != nil {
        RespondError(w, 400, "INVALID_REQUEST", "invalid JSON")
        return
    }
    ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
    defer cancel()
    _, err := s.bus.Request(ctx, ipc.Message{
        Source: "api", Target: "sla", Topic: "sla.add_rule", Payload: rule,
    })
    if err != nil {
        RespondError(w, 500, "ADD_FAILED", err.Error())
        return
    }
    Respond(w, 201, map[string]string{"status": "created", "id": rule.ID})
}

func (s *Server) handleSLAListRules(w http.ResponseWriter, r *http.Request) {
    ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
    defer cancel()
    resp, err := s.bus.Request(ctx, ipc.Message{
        Source: "api", Target: "sla", Topic: "sla.list_rules",
    })
    if err != nil {
        Respond(w, 200, []any{})
        return
    }
    Respond(w, 200, resp.Payload)
}

func (s *Server) handleSLARemoveRule(w http.ResponseWriter, r *http.Request) {
    ruleID := r.PathValue("id")
    ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
    defer cancel()
    _, err := s.bus.Request(ctx, ipc.Message{
        Source: "api", Target: "sla", Topic: "sla.remove_rule", Payload: ruleID,
    })
    if err != nil {
        RespondError(w, 404, "NOT_FOUND", err.Error())
        return
    }
    Respond(w, 200, map[string]string{"status": "removed", "id": ruleID})
}

func (s *Server) handleSLAViolations(w http.ResponseWriter, r *http.Request) {
    ctx, cancel := context.WithTimeout(r.Context(), 5*time.Second)
    defer cancel()
    resp, err := s.bus.Request(ctx, ipc.Message{
        Source: "api", Target: "sla", Topic: "sla.violations",
    })
    if err != nil {
        Respond(w, 200, []any{})
        return
    }
    Respond(w, 200, resp.Payload)
}
```
